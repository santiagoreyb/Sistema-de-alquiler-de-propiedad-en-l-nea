CREATE TABLE Servicio(
id NUMBER(10, 0) GENERATED ALWAYS AS IDENTITY NOT NULL ENABLE,
nombre VARCHAR2(50) NOT NULL ENABLE,
precio NUMBER(10) NOT NULL ENABLE,
numInquilinos NUMBER(3) NOT NULL ENABLE,
costo NUMBER(10) DEFAULT 0 NOT NULL ENABLE,
CONSTRAINT SUK_1 UNIQUE (nombre),
CONSTRAINT SPK_1 PRIMARY KEY (id)
)

CREATE TABLE ServicioPorRenta(
idServicio NUMBER(10) NOT NULL ENABLE,
